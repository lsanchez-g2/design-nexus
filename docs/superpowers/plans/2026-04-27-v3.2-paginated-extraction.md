```javascript
/**
 * Paginated extraction for large files (≥50 pages)
 * Processes pages in batches to avoid timeout, saves partial snapshots, supports resume
 */
async function extractWithPagination(fileKey, pageCount, batchSize) {
  const totalBatches = Math.ceil(pageCount / batchSize);
  const partialDir = `.design-nexus/snapshots/.partial/`;
  
  // Step 1: Check for existing partial snapshots (resume support)
  const existingBatches = await findExistingBatches(fileKey, partialDir);
  const startBatch = existingBatches.length;
  
  if (startBatch > 0) {
    console.log(`📂 Resuming from batch ${startBatch + 1} (found ${startBatch} existing batches)\n`);
  }
  
  // Step 2: Process remaining batches
  for (let batchIndex = startBatch; batchIndex < totalBatches; batchIndex++) {
    const startPage = batchIndex * batchSize;
    const endPage = Math.min((batchIndex + 1) * batchSize - 1, pageCount - 1);
    
    // Extract this batch
    const startTime = Date.now();
    const partial = await extractBatch(fileKey, startPage, endPage, batchIndex);
    const duration = ((Date.now() - startTime) / 1000).toFixed(1);
    
    // Save partial snapshot atomically
    await savePartialSnapshot(partial, partialDir);
    
    // Display progress
    displayBatchProgress(
      batchIndex,
      totalBatches,
      startPage,
      endPage,
      Object.keys(partial.components).length,
      duration
    );
  }
  
  // Step 3: Merge all partial snapshots
  const merged = await mergePartialSnapshots(fileKey, totalBatches, partialDir);
  
  // Step 4: Save final snapshot
  await saveSnapshot(merged, `.design-nexus/snapshots/${fileKey}.json`);
  
  // Step 5: Clean up .partial/ directory
  await cleanupPartialSnapshots(fileKey, partialDir);
  
  return merged;
}
```

```javascript
/**
 * Extract components and variables from a batch of pages
 * @param {string} fileKey - Figma file key
 * @param {number} startPage - First page index in batch
 * @param {number} endPage - Last page index in batch (inclusive)
 * @param {number} batchIndex - Batch number (0-indexed)
 * @returns {Object} Partial snapshot data
 */
async function extractBatch(fileKey, startPage, endPage, batchIndex) {
  const components = {};
  let variables = null;
  const pageNames = [];
  
  // Extract components from pages [startPage...endPage]
  for (let i = startPage; i <= endPage && i < figma.root.children.length; i++) {
    const page = figma.root.children[i];
    
    // Switch to page and load content
    await figma.setCurrentPageAsync(page);
    pageNames.push(page.name);
    
    // Reuse v3.1 component extraction logic
    const pageComponents = await extractPageComponents(page);
    Object.assign(components, pageComponents);
  }
  
  // Extract variables ONCE (file-level, not page-level)
  // Only extract in batch 0 to avoid redundant work
  if (batchIndex === 0) {
    variables = await extractVariables(fileKey);
  }
  
  // Return partial snapshot data
  return {
    version: '3.2.0',
    fileKey,
    fileName: figma.root.name,
    fileUrl: `https://www.figma.com/design/${fileKey}`,
    batchIndex,
    pageRange: {
      start: startPage,
      end: endPage,
      pageNames: pageNames
    },
    timestamp: new Date().toISOString(),
    components,
    variables: variables || { collections: [], variables: [] },
    metadata: {
      componentCount: Object.keys(components).length,
      extractionDuration: 0 // Filled by caller
    }
  };
}
```

```javascript
/**
 * Save partial snapshot atomically (write to temp, then rename)
 * @param {Object} partial - Partial snapshot data
 * @param {string} partialDir - Directory path (.design-nexus/snapshots/.partial/)
 */
async function savePartialSnapshot(partial, partialDir) {
  // Ensure .partial/ directory exists
  await ensureDirectory(partialDir);
  
  const finalPath = `${partialDir}${partial.fileKey}-batch-${partial.batchIndex}.json`;
  const tempPath = `${finalPath}.tmp`;
  
  try {
    // Write to temporary file first
    await writeFile(tempPath, JSON.stringify(partial, null, 2));
    
    // Atomic rename (if interrupted here, temp file is orphaned but no corruption)
    await renameFile(tempPath, finalPath);
  } catch (error) {
    // Clean up temp file on failure
    await deleteFile(tempPath).catch(() => {});
    throw new Error(`Failed to save partial snapshot batch ${partial.batchIndex}: ${error.message}`);
  }
}

/**
 * Helper: Ensure directory exists (create if not)
 */
async function ensureDirectory(dirPath) {
  try {
    await readFile(`${dirPath}.keep`); // Test if directory is accessible
  } catch (error) {
    // Directory doesn't exist, create it
    await mkdir(dirPath);
  }
}
```

```javascript
/**
 * Find existing partial snapshots for resume support
 * Validates each batch file (JSON structure, batch index)
 * Stops at first missing/corrupted batch
 * @param {string} fileKey - Figma file key
 * @param {string} partialDir - Partial snapshots directory
 * @returns {Array} Valid batch objects (empty if no existing batches)
 */
async function findExistingBatches(fileKey, partialDir) {
  const batches = [];
  let batchIndex = 0;
  
  while (true) {
    const path = `${partialDir}${fileKey}-batch-${batchIndex}.json`;
    
    // Check if batch file exists
    try {
      const exists = await fileExists(path);
      if (!exists) {
        // No more batches found
        break;
      }
    } catch (error) {
      // File system error, stop
      break;
    }
    
    // File exists, validate structure
    try {
      const content = await readFile(path);
      const parsed = JSON.parse(content);
      
      // Validate required fields
      if (parsed.version === '3.2.0' && parsed.batchIndex === batchIndex) {
        batches.push(parsed);
        batchIndex++;
      } else {
        // Invalid batch file
        console.warn(
          `⚠️  Invalid batch file at index ${batchIndex}:\n` +
          `   Expected batchIndex=${batchIndex}, got ${parsed.batchIndex || 'undefined'}\n` +
          `   Discarding and restarting from this batch.\n`
        );
        await deleteFile(path);
        break;
      }
    } catch (error) {
      // Corrupted JSON
      console.warn(
        `⚠️  Corrupted JSON at batch ${batchIndex}: ${error.message}\n` +
        `   Discarding and restarting from this batch.\n`
      );
      await deleteFile(path);
      break;
    }
  }
  
  return batches;
}
```

```javascript
/**
 * Merge all partial snapshots into final baseline
 * @param {string} fileKey - Figma file key
 * @param {number} totalBatches - Total batches to merge
 * @param {string} partialDir - Partial snapshots directory
 * @returns {Object} Final merged snapshot
 */
async function mergePartialSnapshots(fileKey, totalBatches, partialDir) {
  console.log(`\n🔗 Merging ${totalBatches} batches...`);
  const startTime = Date.now();
  
  // Initialize merged snapshot structure
  const merged = {
    version: '3.2.0',
    timestamp: new Date().toISOString(),
    figma: {
      fileKey,
      fileName: '',
      fileUrl: ''
    },
    components: {},
    variables: {
      collections: [],
      variables: []
    },
    metadata: {
      extractionMode: 'paginated',
      batchCount: totalBatches
    }
  };
  
  // Read and merge all partial snapshots
  for (let i = 0; i < totalBatches; i++) {
    const partialPath = `${partialDir}${fileKey}-batch-${i}.json`;
    
    try {
      const content = await readFile(partialPath);
      const partial = JSON.parse(content);
      
      // Validate batch index matches
      if (partial.batchIndex !== i) {
        throw new Error(
          `Batch index mismatch: expected ${i}, got ${partial.batchIndex}\n` +
          `Partial snapshot may be corrupted.`
        );
      }
      
      // Merge components (simple object spread)
      // Component IDs are globally unique, no deduplication needed
      Object.assign(merged.components, partial.components);
      
      // Merge variables (only from batch 0)
      if (i === 0) {
        merged.variables = partial.variables;
        merged.figma.fileName = partial.fileName;
        merged.figma.fileUrl = partial.fileUrl;
      }
    } catch (error) {
      throw new Error(
        `Failed to merge batch ${i}:\n` +
        `${error.message}\n\n` +
        `Partial snapshots preserved in:\n` +
        `${partialDir}\n\n` +
        `You can manually inspect or delete them.`
      );
    }
  }
  
  // Calculate final metadata
  merged.metadata.totalComponents = Object.keys(merged.components).length;
  merged.metadata.totalVariables = merged.variables.variables.length;
  merged.metadata.pageCount = totalBatches * 10; // Approximate
  
  const duration = ((Date.now() - startTime) / 1000).toFixed(1);
  console.log(`   ✓ (${duration}s)`);
  
  return merged;
}
```

```javascript
/**
 * Clean up partial snapshots after successful merge
 * Non-fatal: Logs warning if cleanup fails but doesn't fail the sync
 * @param {string} fileKey - Figma file key
 * @param {string} partialDir - Partial snapshots directory
 */
async function cleanupPartialSnapshots(fileKey, partialDir) {
  try {
    let deletedCount = 0;
    
    // Delete all batch files (iterate up to 999 batches, break on first missing)
    for (let i = 0; i < 999; i++) {
      const path = `${partialDir}${fileKey}-batch-${i}.json`;
      
      try {
        const exists = await fileExists(path);
        if (!exists) {
          // No more batches
          break;
        }
        
        await deleteFile(path);
        deletedCount++;
      } catch (error) {
        // Failed to delete this file, log and continue
        console.warn(`   ⚠️  Could not delete ${path}: ${error.message}`);
      }
    }
    
    if (deletedCount > 0) {
      console.log(`🧹 Cleaned up ${deletedCount} partial snapshots\n`);
    }
  } catch (error) {
    // Non-fatal: Log warning but don't fail the sync
    console.warn(
      `⚠️  Warning: Could not clean up partial snapshots:\n` +
      `   ${error.message}\n` +
      `   You can manually delete: ${partialDir}${fileKey}-batch-*.json\n`
    );
  }
```

```javascript
async function callFigmaWithRetry(operation, maxRetries = 3) {
  let lastError;
  
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      const result = await operation();
      return result;
    } catch (error) {
      lastError = error;
      
      // Check if error is retryable
      if (isRetryableError(error)) {
        const delayMs = Math.pow(2, attempt) * 1000; // 2s, 4s, 8s
        console.warn(`⚠️  Figma API error (attempt ${attempt}/${maxRetries}): ${error.message}`);
        
        if (attempt < maxRetries) {
          console.log(`  ↻ Retrying in ${delayMs/1000}s...\n`);
          await sleep(delayMs);
          continue;
        }
      }
      
      throw error;
    }
  }
  
  throw new Error(`Figma API failed after ${maxRetries} attempts: ${lastError.message}`);
}
```
